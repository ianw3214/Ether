cmake_minimum_required(VERSION 3.16)
project(Ether LANGUAGES C CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

add_executable(Ether src/main.cpp)

# Emscripten specific settings
if (EMSCRIPTEN)
    set_target_properties(Ether PROPERTIES
        OUTPUT_NAME "engine"
        SUFFIX ".js")

    # Why is leading underscore necessary for exported funcs?
    target_link_options(Ether PRIVATE
        "-sASSERTIONS=1"
        "-sSAFE_HEAP=1"
        "-sALLOW_MEMORY_GROWTH=1"
        "-sMODULARIZE=1"
        "-sEXPORT_ES6=1"
        "-sENVIRONMENT=web"
        # "-sVERBOSE=1"
        "-sEXPORTED_FUNCTIONS=_main,_engine_init,_engine_tick,_engine_shutdown"
        # "-sEXPORTED_RUNTIME_METHODS=ccall,cwrap"
    )
endif()
